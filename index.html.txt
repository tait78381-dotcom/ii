        // =============================================
        // 8. GHOST TYPING
        // =============================================
        function renderGhost() {
            let html = '';
            const typedLen = typedChars.length;
            const targetLen = targetCode.length;

            for (let i = 0; i < targetLen; i++) {
                if (i < typedLen) {
                    const char = typedChars[i];
                    const target = targetCode[i];
                    if (char === target) {
                        html += '<span class="typed">' + escapeHtml(char) + '</span>';
                    } else {
                        html += '<span class="typed error">' + escapeHtml(char) + '</span>';
                    }
                } else {
                    html += '<span class="ghost">' + escapeHtml(targetCode[i]) + '</span>';
                }
            }
            ghostDisplay.innerHTML = html;
            cursor.style.display = (typedLen < targetLen) ? 'inline-block' : 'none';

            const acc = typedLen > 0 ? getAccuracy() : 0;
            const accDisplay = Math.round(acc);
            matchDisplay.textContent = accDisplay + '%';
            matchDisplay2.textContent = accDisplay + '% khớp';
            accuracyDisplay.textContent = accDisplay;
            updateChecklist();
            updatePreview();
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

        function getAccuracy() {
            if (typedChars.length === 0) return 0;
            let correct = 0;
            const len = Math.min(typedChars.length, targetCode.length);
            for (let i = 0; i < len; i++) {
                if (typedChars[i] === targetCode[i]) correct++;
            }
            return (correct / typedChars.length) * 100;
        }

            } catch (e) {
                previewRender.innerHTML = '<span style="color:#ef4444;">Lỗi preview</span>';
            }
        }

                raw = raw.slice(0, targetCode.length);

            typedChars = raw;

            const range = document.createRange();
            const sel = window.getSelection();
            if (this.childNodes.length > 0) {
                const last = this.childNodes[this.childNodes.length - 1];
                if (last.nodeType === 3) {
                    range.setStart(last, last.textContent.length);
                } else {
                    range.setStart(this, this.childNodes.length);
                }
                range.collapse(true);
                sel.removeAllRanges();
                sel.addRange(range);
            }
            renderGhost();
            if (typedChars === targetCode) {
                isCompleted = true;
                checkBtn.disabled = false;
                showToast('✅ Hoàn thành! Bấm KIỂM TRA để nhận XP.', 1500);

            if (c1 && c2 && acc >= 90) {
                if (state.completed[state.currentRealm] <= state.currentLessonIndex) {
                    state.xp += 50;
                    state.completed[state.currentRealm]++;
                    state.level = Math.floor(state.xp / 220) + 1;
                }
                checkBtn.disabled = true;
                saveState();
                updateUI();
                showToast('🎉 Hoàn thành! +50 XP', 1500);

                if (state.currentLessonIndex < lessons.length - 1) {
                    setTimeout(() => {
                        state.currentLessonIndex++;
                        loadLesson(state.currentRealm, state.currentLessonIndex);
                    }, 800);
                } else {
                    showToast('🏆 Bạn đã hoàn thành tất cả bài học trong realm này!', 3000);
                }
            } else {
                showToast('❌ Chưa đạt yêu cầu. Kiểm tra lại code.', 1500);
            }
        }

        function showHint() {
            const lessons = LESSONS_DATA[state.currentRealm];
            if (!lessons || state.currentLessonIndex >= lessons.length) return;
            const lesson = lessons[state.currentLessonIndex];
            if (!lesson) return;
            showToast('💡 Gợi ý: ' + lesson.code.substring(0, 30) + '...', 2500);
            state.xp += 10;
            saveState();
            updateUI();
        }

        function clearInput() {
            typedChars = '';
            ghostBox.innerText = '';
            renderGhost();
            showToast('🔄 Đã xóa, gõ lại.', 1500);
            ghostBox.focus();
        }

        function autoType() {
            if (typedChars.length < targetCode.length) {
                typedChars += targetCode[typedChars.length];
                ghostBox.innerText = typedChars;
                renderGhost();
                ghostBox.focus();
                const range = document.createRange();
                const sel = window.getSelection();
                range.setStart(ghostBox.childNodes[0] || ghostBox, typedChars.length);
                range.collapse(true);
                sel.removeAllRanges();
                sel.addRange(range);
            }
        }

        // =============================================
        // 13. MAP
        // =============================================
        function renderMap(realm) {
            const lessons = LESSONS_DATA[realm];
            const grid = document.getElementById('mapGrid');
            grid.innerHTML = '';
            const completed = state.completed[realm] || 0;
            const total = lessons.length;

            for (let i = 0; i < total; i++) {
                const node = document.createElement('div');
                node.className = 'map-node';
                if (i < completed) {
                    node.classList.add('completed');
                    node.textContent = '✓';
                } else if (i <= completed) {
                    node.classList.add('unlocked');
                    if (i === completed) node.classList.add('current');
                    node.textContent = i + 1;
                } else {
                    node.classList.add('locked');
                    node.textContent = '🔒';
                }
                node.onclick = function() {
                    if (i <= completed) {
                        state.currentLessonIndex = i;
                        loadLesson(realm, i);
                    } else {
                        showToast('🔒 Bài chưa mở khóa!', 1500);
                    }
                };
                grid.appendChild(node);
            }
        }

        // =============================================
        // 14. TOAST
        // =============================================
        function showToast(msg, duration) {
            duration = duration || 2500;
            const el = document.getElementById('toast');
            el.textContent = msg;
            el.classList.add('show');
            clearTimeout(el._timer);
            el._timer = setTimeout(() => el.classList.remove('show'), duration);
        }

        // =============================================
        // 15. KHỞI TẠO
        // =============================================
        loadState();
    </script>
